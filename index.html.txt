      overlay.classList.toggle('hidden', !show);
      drawer.classList.toggle('hidden', !show);
      setTimeout(()=> drawer.classList.toggle('show', show), 0);
    }

    function updateCartUI(){
      const list = Object.entries(state.cart).map(([id, qty])=> ({...MENU.find(m=>m.id===id), qty}));
      $('#cartCount').textContent = list.reduce((s,i)=>s+i.qty,0);
      const subtotal = list.reduce((s,i)=> s + i.price*i.qty, 0);
      const tax = subtotal * 0.0725; const total = subtotal + tax;
      $('#subtotal').textContent = money(subtotal);
      $('#tax').textContent = money(tax);
      $('#total').textContent = money(total);
      $('#payTotal').textContent = money(total);
      $('#checkout').disabled = list.length===0;

      const rows = list.map(i => `
        <div class="flex items-start justify-between gap-3 border-b pb-3">
          <div>
            <div class="font-medium">${i.name}</div>
            <div class="text-sm text-slate-500">${money(i.price)} ‚Ä¢ ${i.category}</div>
            <div class="mt-2 inline-flex items-center gap-2">
              <label class="text-xs text-slate-500">Qty</label>
              <input type="number" min="0" value="${i.qty}" data-qty="${i.id}" class="w-16 rounded-lg border border-slate-300 px-2 py-1"/>
            </div>
          </div>
          <div class="text-right">
            <div class="font-semibold">${money(i.price*i.qty)}</div>
            <button data-remove="${i.id}" class="text-xs text-rose-600 hover:underline mt-1">Remove</button>
          </div>
        </div>`).join('');
      $('#cartItems').innerHTML = rows || '<p class="text-slate-600">Your cart is empty. Add some pies!</p>';
    }

    // Qty & remove
    document.addEventListener('input', (e)=>{
      const el = e.target.closest('[data-qty]');
      if(!el) return;
      const id = el.getAttribute('data-qty');
      const val = Math.max(0, Number(el.value||0));
      if(val<=0) delete state.cart[id]; else state.cart[id]=val;
      updateCartUI();
    });
    document.addEventListener('click', (e)=>{
      const r = e.target.closest('[data-remove]');
      if(!r) return;
      const id = r.getAttribute('data-remove');
      delete state.cart[id];
      updateCartUI();
    });

    // Clear cart
    $('#clear').onclick = ()=>{ state.cart = {}; updateCartUI(); };

    // Checkout modal
    const modal = $('#modal');
    const modalOverlay = $('#modalOverlay');
    $('#checkout').onclick = ()=>{ modal.classList.remove('hidden'); modalOverlay.classList.remove('hidden'); };
    modalOverlay.onclick = ()=>{ modal.classList.add('hidden'); modalOverlay.classList.add('hidden'); };
    $('#pay').onsubmit = (e)=>{ e.preventDefault(); alert('Order placed! See you soon.'); modal.classList.add('hidden'); modalOverlay.classList.add('hidden'); toggleCart(false); state.cart={}; updateCartUI(); };

    // Forms
    $('#contactForm').onsubmit = (e)=>{ e.preventDefault(); alert('Message sent! We‚Äôll reply soon.'); };
    $('#newsletter').onsubmit = (e)=>{ e.preventDefault(); alert('Subscribed! Check your inbox.'); };

    // Footer year
    $('#year').textContent = new Date().getFullYear();

    // Initialize
    updateCartUI();
